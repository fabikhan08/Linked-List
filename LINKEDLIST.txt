class Node:

        self.value=value
        self.next=None
class ListStack:
    def __init__(self,size):
        self.size=size
        self.top=0
        self.head=None
        self.tail=None
    def isEmpty(self):
        if self.top==0:
            return True
        else:
            return False
    def Push(self,value):
        if self.top==self.size:
            print("Stack Overflow !")
        else:
            self.top+=1
            newnode=Node(value)
            if self.head == None:
                self.head=newnode
                self.tail=newnode
            else:
                newnode.next=self.head
                self.head=newnode
    def Pop (self):
        if self.isEmpty():
            print("Stack Underflow !")
        else:
            self.top-=1
            x=self.head
            x=x.next
            self.head=x
    def Check(self):
        if self.isEmpty:
            True
        else:
            False
    def Peek(self):
        print("Peek value of stack is:",self.head.value)
    def Count(self):
        return "Number of elements in stack: "+str(self.top)
    def Print(self):
        x=self.head
        print("\nTop to down.")
        while x:
            print("Stack:|_",x.value,"_|")
            x=x.next
ob=ListStack(3)
